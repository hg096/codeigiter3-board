CREATE TABLE `user` (
	`u_idx` INT(10) NOT NULL AUTO_INCREMENT,
	`u_id` VARCHAR(100) NULL DEFAULT NULL COLLATE ,
	`u_email` VARCHAR(100) NULL DEFAULT NULL COLLATE ,
	`u_pw` VARCHAR(255) NULL DEFAULT NULL COLLATE ,
	`u_date` DATETIME NULL DEFAULT NULL,
	PRIMARY KEY (`u_idx`) USING BTREE
)
ENGINE=InnoDB

;


CREATE TABLE `board` (
	`b_idx` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`b_name` VARCHAR(100) NULL DEFAULT ,
	`b_pw` VARCHAR(255) NULL DEFAULT NULL ,
	`b_title` VARCHAR(100) NULL DEFAULT ,
	`b_content` TEXT NULL DEFAULT NULL ,
	`b_date` DATETIME NULL DEFAULT NULL,
	`b_hit` INT(10) UNSIGNED NULL DEFAULT '0',
	`b_lock_post` INT(10) UNSIGNED NULL DEFAULT '0',
	`b_file` VARCHAR(100) NULL DEFAULT ,
	PRIMARY KEY (`b_idx`) USING BTREE
)
ENGINE=InnoDB
;

CREATE TABLE `reply` (
	`r_idx` INT(10) NOT NULL AUTO_INCREMENT,
	`r_board_idx` INT(10) NULL DEFAULT NULL,
	`r_name` VARCHAR(100) NULL DEFAULT ,
	`r_pw` VARCHAR(100) NULL DEFAULT ,
	`r_content` TEXT NULL DEFAULT NULL COLLATE ,
	`r_date` DATETIME NULL DEFAULT NULL,
	PRIMARY KEY (`r_idx`) USING BTREE
)
ENGINE=InnoDB
;
